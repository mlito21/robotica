🗓️Fase #4

💻Programación movimiento y audio

#include <SoftwareSerial.h>
#include "DFRobotDFPlayerMini.h"
int VELOCIDAD;


SoftwareSerial DFPlayerSerial (8,7);
DFRobotDFPlayerMini myDFPlayer;


void setup() {
 
 
  pinMode (13, OUTPUT);
  pinMode (12,OUTPUT);
  pinMode (11, OUTPUT);
  pinMode (10, OUTPUT);
  pinMode (6, OUTPUT);

  DFPlayerSerial.begin (9600);
  myDFPlayer.begin (DFPlayerSerial);
  myDFPlayer.volume (23);
  myDFPlayer.play ();
 


}

void loop() {
  digitalWrite (6, HIGH);


   //VELOCIDAD
  VELOCIDAD = 100;
//MOVIMIENTO HACIA ATRAS
  aMovimientoDelante (VELOCIDAD);
  delay (3000);  
  aMovimientoAtras (VELOCIDAD);
  delay (6000);
  aMovimientoDelante (VELOCIDAD);
  delay (3000);
 
}

void aMovimientoAtras (int veloc){
  analogWrite(13, HIGH);
  digitalWrite(12, LOW);
  digitalWrite(11, HIGH);
  digitalWrite(10, LOW);

}
void aMovimientoDelante (int veloc){
  analogWrite (13, LOW);
  digitalWrite (12, HIGH);
  digitalWrite (11, LOW);
  digitalWrite (10, HIGH);
  }
